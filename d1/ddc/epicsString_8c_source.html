330  *lasts = NULL;

336  * Scan token (scan for delimiters: s += strcspn(s, delim), sort of).

338  */
339  for (;;) {

343  if ((sc = *spanp++) == c) {
344  if (c == 0)
345  s = NULL;

348  *lasts = s;

350  }
351  } while (sc != 0);
