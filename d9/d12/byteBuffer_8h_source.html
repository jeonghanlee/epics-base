116 struct swap<4> {
117  static EPICS_ALWAYS_INLINE uint32 op(uint32 v) {
118 #ifdef _PVA_swap32
119  return _PVA_swap32(v);
120 #else
121  return ((((v) & 0xff000000) >> 24) |
122  (((v) & 0x00ff0000) >> 8) |
123  (((v) & 0x0000ff00) << 8) |
124  (((v) & 0x000000ff) << 24));
125 #endif
126  }

133  }
134 #else
135  static inline uint64 op(uint64 v) {
136  return (((v) >> 56) | \
137  (((v) >> 40) & 0x0000ff00) | \
138  (((v) >> 24) & 0x00ff0000) | \
139  (((v) >> 8) & 0xff000000) | \

141  (((v) << 24) & ((uint64_t)0xff << 40)) | \
