81 epicsReadline (const char *prompt, void *context)
82 {
83  struct readlineContext *rc = context;
84  FILE *in;
85  char *line;
86  int c; /* char is unsigned on some archs, EOF is -ve */
87  int linelen = 0;
88  int linesize = 50;
89 
90  if (rc->osd)
91  return osdReadline(prompt, rc);
92 
93  free(rc->line);
94  rc->line = NULL;
95  if ((in = rc->in) == NULL) {
96  in = stdin;
97  if (prompt) {
98  fputs(prompt, stdout);
99  fflush(stdout);
100  }
101  }
102  line = (char *)malloc(linesize);
103  if (line == NULL) {
104  printf("Out of memory!\n");
105  return NULL;
106  }
107  while ((c = getc(in)) != '\n') {
108  if (c == EOF) {
109  if (ferror(in)) {
110  if ((errno == EINTR) || (errno == EPIPE)) {
111  clearerr(in);
112  continue;
113  }
114  }
115  free (line);
116  return NULL;
117  }
118  if ((linelen + 1) >= linesize) {
119  char *cp;
120 
121  linesize += 50;
122  cp = (char *)realloc(line, linesize);
123  if (cp == NULL) {
124  printf("Out of memory!\n");
125  free(line);
126  return NULL;
127  }
128  line = cp;
129  }
130  line[linelen++] = c;
131  }
132  line[linelen] = '\0';
133  rc->line = line;
134  return line;
135 }
