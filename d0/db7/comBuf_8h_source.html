311 template < class T >

313 {
314  unsigned nrIndex = this->nextReadIndex;
315  unsigned popIndex = nrIndex + sizeof ( returnVal );
316  unsigned cIndex = this->commitIndex;
317  popStatus status;

328  }

330  this->nextReadIndex = popIndex;
