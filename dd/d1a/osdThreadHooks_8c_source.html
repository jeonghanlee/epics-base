77 LIBCOM_API int epicsThreadHookDelete(EPICS_THREAD_HOOK_ROUTINE hook)
78 {
79  if (!hook) return 0;

82  if (epicsMutexLock(hookLock) == epicsMutexLockOK) {
83  epicsThreadHook *pHook = (epicsThreadHook *) ellFirst(&hookList);
84
