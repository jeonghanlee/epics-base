165  * (and the replacements are not standardized)
166  */
167  status = epicsSnprintf (

169  chunk[3], chunk[2], chunk[1], chunk[0],
170  ntohs ( paddr->sin_port ) );
171  if ( status > 0 ) {
172  strLen = ( unsigned ) status;

181  }
182  else {

184  pBuf[bufSize-1] = '\0';
185  return bufSize - 1u;
