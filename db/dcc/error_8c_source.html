47 static int
48 print_pos(char *st_line, char *st_cptr)
49 {
50  char *s;
51 
52  if (st_line == 0) return(0);
53  for (s = st_line; *s != '\n'; ++s)
54  {
55  if (isprint((int) *s) || *s == '\t')
56  putc(*s, stderr);
57  else
58  putc('?', stderr);
59  }
60  putc('\n', stderr);
61  for (s = st_line; s < st_cptr; ++s)
62  {
63  if (*s == '\t')
64  putc('\t', stderr);
65  else
66  putc(' ', stderr);
67  }
68  putc('^', stderr);
69  putc('\n', stderr);
70  return(0);
71 }
