35 void TimeStamp::normalize()
36 {
37  if(nanoseconds>=0 && nanoseconds<nanoSecPerSec) return;
38  while(nanoseconds>=nanoSecPerSec) {
39  nanoseconds -= nanoSecPerSec;
40  secondsPastEpoch++;
41  }
42  while(nanoseconds<0) {
43  nanoseconds += nanoSecPerSec;
44  secondsPastEpoch--;
45  }
46 }
